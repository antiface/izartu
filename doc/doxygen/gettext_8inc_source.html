00066 function get_list_of_locales($locale) {
00067   /* Figure out all possible locale names and start with the most
00068    * specific ones.  I.e. for sr_CS.UTF-8@latin, look through all of
00069    * sr_CS.UTF-8@latin, sr_CS@latin, sr@latin, sr_CS.UTF-8, sr_CS, sr.
00070    */
00071   $locale_names = array();
00072   $lang = NULL;
00073   $country = NULL;
00074   $charset = NULL;
00075   $modifier = NULL;
00076   if ($locale) {
00077     if (preg_match("/^(?P<lang>[a-z]{2,3})"              // language code
00078                    ."(?:_(?P<country>[A-Z]{2}))?"           // country code
00079                    ."(?:\.(?P<charset>[-A-Za-z0-9_]+))?"    // charset
00080                    ."(?:@(?P<modifier>[-A-Za-z0-9_]+))?$/",  // @ modifier
00081                    $locale, $matches)) {
00082 
00083       if (isset($matches["lang"])) $lang = $matches["lang"];
00084       if (isset($matches["country"])) $country = $matches["country"];
00085       if (isset($matches["charset"])) $charset = $matches["charset"];
00086       if (isset($matches["modifier"])) $modifier = $matches["modifier"];
00087 
00088       if ($modifier) {
00089         if ($country) {
00090           if ($charset)
00091             array_push($locale_names, "${lang}_$country.$charset@$modifier");
00092           array_push($locale_names, "${lang}_$country@$modifier");
00093         } elseif ($charset)
00094             array_push($locale_names, "${lang}.$charset@$modifier");
00095         array_push($locale_names, "$lang@$modifier");
00096       }
00097       if ($country) {
00098         if ($charset)
00099           array_push($locale_names, "${lang}_$country.$charset");
00100         array_push($locale_names, "${lang}_$country");
00101       } elseif ($charset)
00102           array_push($locale_names, "${lang}.$charset");
00103       array_push($locale_names, $lang);
00104     }
00105 
00106     // If the locale name doesn't match POSIX style, just include it as-is.
00107     if (!in_array($locale, $locale_names))
00108       array_push($locale_names, $locale);
00109   }
00110   return $locale_names;
00111 }

00116 function _get_reader($domain=null, $category=5, $enable_cache=true) {
00117     global $text_domains, $default_domain, $LC_CATEGORIES;
00118     if (!isset($domain)) $domain = $default_domain;
00119     if (!isset($text_domains[$domain]->l10n)) {
00120         // get the current locale
00121         $locale = _setlocale(LC_MESSAGES, 0);
00122         $bound_path = isset($text_domains[$domain]->path) ?
00123           $text_domains[$domain]->path : './';
00124         $subpath = $LC_CATEGORIES[$category] ."/$domain.mo";
00125 
00126         $locale_names = get_list_of_locales($locale);
00127         $input = null;
00128         foreach ($locale_names as $locale) {
00129           $full_path = $bound_path . $locale . "/" . $subpath;
00130           if (file_exists($full_path)) {
00131             $input = new FileReader($full_path);
00132             break;
00133           }
00134         }
00135 
00136         if (!array_key_exists($domain, $text_domains)) {
00137           // Initialize an empty domain object.
00138           $text_domains[$domain] = new domain();
00139         }
00140         $text_domains[$domain]->l10n = new gettext_reader($input,
00141                                                           $enable_cache);
00142     }
00143     return $text_domains[$domain]->l10n;

00203 function _setlocale($category, $locale) {
00204     global $CURRENTLOCALE, $EMULATEGETTEXT;
00205     if ($locale === 0) { // use === to differentiate between string "0"
00206         if ($CURRENTLOCALE != '')
00207             return $CURRENTLOCALE;
00208         else
00209             // obey LANG variable, maybe extend to support all of LC_* vars
00210             // even if we tried to read locale without setting it first
00211             return _setlocale($category, $CURRENTLOCALE);
00212     } else {
00213         if (function_exists('setlocale')) {
00214           $ret = setlocale($category, $locale);
00215           if (($locale == '' and !$ret) or // failed setting it by env
00216               ($locale != '' and $ret != $locale)) { // failed setting it
00217             // Failed setting it according to environment.
00218             $CURRENTLOCALE = _get_default_locale($locale);
00219             $EMULATEGETTEXT = 1;
00220           } else {
00221             $CURRENTLOCALE = $ret;
00222             $EMULATEGETTEXT = 0;
00223           }
00224         } else {
00225           // No function setlocale(), emulate it all.
00226           $CURRENTLOCALE = _get_default_locale($locale);
00227           $EMULATEGETTEXT = 1;
00228         }
00229         // Allow locale to be changed on the go for one translation domain.
00230         global $text_domains, $default_domain;
00231         if (array_key_exists($default_domain, $text_domains)) {
00232             unset($text_domains[$default_domain]->l10n);
00233         }
00234         return $CURRENTLOCALE;
00235     }
